
defined_text = { 
	name = GetITADomesticOpp
	text = {
		trigger = {
			check_variable = {
                var = ita_var_domestic_opposition
                value = 0.60 
                compare = greater_than_or_equals 
            }
		}
		localization_key = "ITA_HIGH_OPPOSITION"
	}
	text = {
		trigger = {
			check_variable = { ita_var_domestic_opposition < 0.60 }
			check_variable = { ita_var_domestic_opposition > 0.30 }
		}
		localization_key = "ITA_MEDIUM_OPPOSITION"
	}
	text = {
		trigger = {
			check_variable = {
                var = ita_var_domestic_opposition
                value = 0.30  
                compare = less_than_or_equals  
            }
		}
		localization_key = "ITA_LOW_OPPOSITION"
	}
}
